I. บทที่ 12.4   addEventListener()

EventListener เป็นเหมือนตัวรับสัญญาณ 
รับสัญญาณ ก่อน ค่อยทำอะไร 


EX
เเบบที่1
// v3.addEventListener("click", function () {
//     alert('Hello V3')
// })

เเบบที่2
// v3.addEventListener("click", () => {
//     alert('Hello V3')
// })

เเบบที่3
function clicked() {
    alert('Hello V3')
}

v3.addEventListener("click", clicked)
