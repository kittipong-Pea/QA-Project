const maxnumber = prompt("Please Enter")
const targetnum =   Math.floor(Math.random()*maxnumber)

console.log(targetnum, "<< targetnum");


let guess = prompt("Enter your first anser")
let attempts = 1 ;

while(guess != targetnum){
    if (guess == "q"){
        break;
    }
    attempts += 1

    if(guess > targetnum){
        guess = prompt("Your Number too high anser")
    } else if (guess < targetnum){
        guess = prompt("Your Number too low anser")
    }
}

if(guess == "q"){
    alert('Ok U out');
}else{
    alert(`You got it ${attempts} guess`);    (`อันนี้คือตัวที่ตรงเเปลภาษานะ)
}
