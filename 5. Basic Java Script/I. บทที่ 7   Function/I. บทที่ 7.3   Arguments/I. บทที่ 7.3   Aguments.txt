I. บทที่ 7.3   Arguments

คือ input หนึ่งที่ใส่ไว้ใน function


EX
เเบบไม่มี Argument

function hello() {
    console.log("Hi");
}
hello();


เเบบมี Argument
function hello(name) {
    console.log(`Hi ${name}`);
}

hello("Tim");
hello("anna");
hello("first");

EX2
function test(num) {
    console.log(num *2);
    
}

test(1);
test(5);
